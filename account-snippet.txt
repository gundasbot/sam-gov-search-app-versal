const handlePlanChange = async (newTier: PlanTier) => {
  if (!plan || !session) return
  
  const isCurrentPlan = plan.tier === newTier && 
                       plan.interval === (billingInterval === 'annual' ? 'year' : 'month')
  
  if (isCurrentPlan) {
    setMessage({ type: 'error', text: 'This is already your current plan' })
    setTimeout(() => setMessage(null), 3000)
    return
  }

  try {
    setLoadingPlanChange(newTier)
    setMessage(null)

    // Check if user is in trial - use the same API as pricing page
    if (plan.subscriptionId && plan.status?.toLowerCase() === 'trialing') {
      // User is in trial - use update-subscription API
      console.log('User in trial, using update-subscription API for:', newTier, billingInterval)
      
      const res = await fetch('/api/stripe/update-subscription', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          tier: newTier,
          interval: billingInterval,
          isDowngrade: false, // Assume upgrade for account page
        }),
      })

      if (!res.ok) {
        const data = await res.json()
        throw new Error(data.error || 'Failed to update subscription')
      }

      const updateData = await res.json()
      
      setMessage({ 
        type: 'success', 
        text: updateData.message || `Plan changed to ${PLAN_FEATURES[newTier].name}!` 
      })
      
      // Reload plan data
      await loadPlan()
      await loadUsage()
      
      setTimeout(() => setMessage(null), 5000)
    }
    else if (!plan.subscriptionId || plan.tier === 'NONE') {
      // NEW USER or NO SUBSCRIPTION - Create checkout
      console.log('Creating new checkout for:', newTier, billingInterval)
      
      const res = await fetch('/api/stripe/checkout', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          plan: newTier.toLowerCase(),
          billing: billingInterval,
        }),
      })

      if (!res.ok) {
        const data = await res.json()
        throw new Error(data.error || 'Failed to create checkout session')
      }

      const { url } = await res.json()
      if (url) {
        // Redirect to Stripe Checkout
        window.location.href = url
      } else {
        throw new Error('No checkout URL received')
      }
    } else {
      // EXISTING USER WITH ACTIVE SUBSCRIPTION - Use change-plan API
      console.log('Changing existing subscription:', newTier, billingInterval)
      
      const res = await fetch('/api/stripe/change-plan', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          plan: newTier.toLowerCase(),
          billing: billingInterval,
        }),
      })

      if (!res.ok) {
        const data = await res.json()
        throw new Error(data.error || 'Failed to change subscription')
      }

      setMessage({ 
        type: 'success', 
        text: `Plan changed to ${PLAN_FEATURES[newTier].name}. Changes will take effect immediately.` 
      })
      
      // Reload plan data
      await loadPlan()
      await loadUsage()
      
      setTimeout(() => setMessage(null), 5000)
    }
  } catch (error: any) {
    console.error('Plan change error:', error)
    setMessage({ 
      type: 'error', 
      text: error.message || 'Failed to change plan. Please try again.' 
    })
    setTimeout(() => setMessage(null), 5000)
  } finally {
    setLoadingPlanChange(null)
  }
}
